import pandas as pd
import matplotlib.pyplot as plt
import numpy as np

# Loading the dataset AND CONVERTING STREAMS INTO NUMERIC VALUE CAUSE OF PRESENCE OF 'E'
data = pd.read_csv('spotify-2023.csv', encoding='ISO-8859-1')
data['streams'] = pd.to_numeric(data['streams'], errors='coerce')
data_cleaned = data.dropna(subset=['streams'])
top_streamed = data_cleaned.nlargest(100, 'streams')
# Attaining the danceability and energy columns
danceability = top_streamed['danceability_%'].astype(float)
energy = top_streamed['energy_%'].astype(float)
# Calculating correlation between danceability and energy
correlation = np.corrcoef(danceability, energy)[0, 1]

# Plotting the relationship between danceability and energy
plt.figure(figsize=(8, 6))
plt.scatter(danceability, energy, color='skyblue', alpha=0.7)
plt.title('Relationship between Danceability and Energy Levels')
plt.xlabel('Danceability (%)')
plt.ylabel('Energy (%)')
plt.text(0.1, 0.9, f'Correlation: {correlation:.2f}', ha='center', va='center', transform=plt.gca().transAxes)
plt.grid(True)
plt.tight_layout()
plt.show()

As we can see the correlation between Danceability and Energy Levels is 0.12 as attained from the code above wherein I had taken the top streamed danceable songs and top streamed energy levels song and from the graph attained we can conclude that:

  • There is a relatively weak correlation between the two variables being analyzed. There may be some relationship between the variables, but it is not a strong or clear linear relationship but a very weak positive correlation between the two.

import pandas as pd
import matplotlib.pyplot as plt

# Load the dataset
data = pd.read_csv('spotify-2023.csv', encoding='ISO-8859-1')

# Clean the 'in_shazam_charts' column by removing commas and converting to numeric
data['in_shazam_charts'] = data['in_shazam_charts'].str.replace(',', '').astype(float)

# Filter songs present on Shazam charts
shazam_songs = data[data['in_shazam_charts'].notnull()]

# Plotting relationship between speechiness and Shazam charts presence
plt.figure(figsize=(8, 6))
plt.scatter(shazam_songs['speechiness_%'], shazam_songs['in_shazam_charts'])
plt.xlabel('Speechiness (%)')
plt.ylabel('Presence on Shazam Charts')
plt.title('Relationship between Speechiness and Shazam Charts Presence')
plt.grid(True)
plt.show()

# Calculating the correlation coefficient
correlation_coeff = shazam_songs['speechiness_%'].corr(shazam_songs['in_shazam_charts'])
print(f"Correlation Coefficient between Speechiness and Shazam Charts Presence: {correlation_coeff:.2f}")

Correlation Coefficient between Speechiness and Shazam Charts Presence: -0.06

We have found the correlation coefficient between the two variables to be -0.06 which is close to 0 and implies a very weak negative linear relationship between the variables. There might be a slight tendency for one variable to increase as the other decreases but the relationship is so week that it would not be easily predictable purely based on this correlation alone.

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Load the dataset with a different encoding (e.g., ISO-8859-1)
data = pd.read_csv('spotify-2023.csv', encoding='ISO-8859-1')

# Filtering songs that appear in at least one of the specified charts (e.g., Spotify, Apple Music, Deezer)
chart_columns = ['in_spotify_charts', 'in_apple_charts', 'in_deezer_charts'] 
chart_songs = data[data[chart_columns].notnull().any(axis=1)]

# Converting 'bpm' column to numeric since it is not(Cleaning)
chart_songs['bpm'] = pd.to_numeric(chart_songs['bpm'], errors='coerce')

# Calculating average bpm for songs in charts
avg_bpm_chart = chart_songs.groupby(chart_columns)['bpm'].mean().reset_index()

# Plotting the correlation between bpm and charts presence
plt.figure(figsize=(10, 6))
for chart in chart_columns:
    plt.scatter(avg_bpm_chart[chart], avg_bpm_chart['bpm'], label=chart)

plt.xlabel('Chart Presence')
plt.ylabel('Average BPM')
plt.title('Correlation between BPM and Chart Presence')
plt.legend()
plt.grid(True)
plt.show()

This is a very conclusive scatter plot as we can see an evident difference among three different streaming platforms with regards to their chart presence and bpm. The conclusions are as follows:

  • In Deezer regardless of BPM, we can see that the chart presence remains considerably low.
  • In Spotift charts, we can see that there is no correlation between average BPM and chart presence however the chart presence on average for songs streamed on spotify is higher than that of Deezer and lower than that of Apple.
  • In apple charts, once again there is no correlation between average BPM and chart presence, however, chart presence on average for songs streamed on apple music is higher than both that of Deezer and Spotify.
